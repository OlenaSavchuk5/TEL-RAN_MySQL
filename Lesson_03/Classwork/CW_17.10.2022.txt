CREATE TABLE Users (
	id integer PRIMARY KEY,
  	first_name varchar(128),
  	last_name varchar(128),
  	username varchar(128),
  	age integer CHECK(age >= 18),
  	gender char(1) CHECK(gender in ('M', 'F'))
);
